== Install

[source,bash]
----
ansible-playbook mariadb/podman/ansible/podman-install-playbook.yaml \
  -e pod_name= \
  -e db_name= \
  -e db_user= \
  -e "@<app>/k8s/ansible/defaults/main.yaml" \
  -e db_password=${DB_PASSWORD}
----


== Uninstall

[source,bash]
----
ansible-playbook mariadb/podman/ansible/podman-uninstall-playbook.yaml \
  -e pod_name=<pod name>
----
